from __future__ import annotations

import time
import datetime
from typing import Callable, Any, Dict, Optional
from functools import wraps

import pandas as pd
import netting_util.NSCMCommon as common

DF = pd.DataFrame

# ---------------------------------------------------------------------
# logger / output_dir cache  (str_instance 기준으로 재사용)
# ---------------------------------------------------------------------
_LOGGER_CACHE: Dict[str, Any] = {}
_OUTPUT_DIR_CACHE: Dict[str, str] = {}


def _get_str_instance_from_module(func: Callable) -> str:
    """
    func.__module__ 기준으로만 판단한다. (요청 조건 준수)

    - func.__module__ 이 'daily_netting...' 로 시작하면:  PYNettingODSD
    - func.__module__ 이 'weekly_netting...' 로 시작하면: PYNettingODSW
    - 그 외: 기본값 PYNettingODSD
    """
    mod = getattr(func, "__module__", "") or ""
    if mod.startswith("daily_netting"):
        return "PYNettingODSD"
    if mod.startswith("weekly_netting"):
        return "PYNettingODSW"
    return "PYNettingODSD"


def _get_logger(str_instance: str):
    if str_instance not in _LOGGER_CACHE:
        _LOGGER_CACHE[str_instance] = common.G_Logger(p_py_name=str_instance)
    return _LOGGER_CACHE[str_instance]


def _get_output_dir(str_instance: str, func: Callable) -> str:
    """
    output_dir 규칙(요청사항):
    - func.__module__ 이 'daily_netting.pre_process.MX....' 형태면
      3번째 토큰('MX')를 str_instance 뒤에 붙여서 'PYNettingODSD_MX' 형태로 만든다.
    - logger는 str_instance 기반으로 그대로 유지 (여기서는 건드리지 않음)
    - 날짜는 YYYYMMDD
    - 캐시는 output_dir_key(str_instance[_SUFFIX]) 단위로 관리
    """
    mod = getattr(func, "__module__", "") or ""
    parts = mod.split(".")
    suffix = parts[2] if len(parts) >= 3 and parts[2] else ""  # 3번째 토큰

    output_dir_key = f"{str_instance}_{suffix}" if suffix else str_instance

    if output_dir_key not in _OUTPUT_DIR_CACHE:
        formatted_time = datetime.datetime.now().strftime("%Y%m%d")
        _OUTPUT_DIR_CACHE[output_dir_key] = f"Output/{output_dir_key}_{formatted_time}"

    return _OUTPUT_DIR_CACHE[output_dir_key]

def _get_output_dir_version(str_instance: str, version: str) -> str:
    """
    output_dir 규칙:
    str_instance + version + formatted_time
    """

    output_dir_key = f"{str_instance}_{version}" if version else str_instance

    if output_dir_key not in _OUTPUT_DIR_CACHE:
        formatted_time = datetime.datetime.now().strftime("%Y%m%d")
        _OUTPUT_DIR_CACHE[output_dir_key] = f"Output/{output_dir_key}_{formatted_time}"

    return _OUTPUT_DIR_CACHE[output_dir_key]

def fn_log_dataframe(df_p_source: DF, str_p_source_name: str, *, logger, str_output_dir: str, is_local: bool) -> None:
    """
    Dataframe 로그 출력 조건 지정 함수 (기존 패턴 유지)
    """
    is_output = str_p_source_name.startswith("out_")
    is_print = True
    flag_csv = True

    if is_print:
        logger.PrintDF(
            p_df=df_p_source,
            p_df_name=str_p_source_name,
            p_log_level=common.G_log_level.debug(),
            p_format=1,
            p_row_num=20,
        )
        if is_local and flag_csv:
            # 로컬 Debugging 시 csv 파일 출력
            try:
                df_p_source.to_csv(
                    str_output_dir + "/" + str_p_source_name + ".csv",
                    encoding="UTF8",
                    index=False,
                )
            except Exception:
                # 로컬 csv 실패는 무시
                pass
    else:
        if is_output:
            logger.PrintDF(
                p_df=df_p_source,
                p_df_name=str_p_source_name,
                p_log_level=common.G_log_level.debug(),
                p_format=1,
                p_row_num=20,
            )
            if is_local and not df_p_source.empty:
                try:
                    df_p_source.to_csv(
                        str_output_dir + "/" + str_p_source_name + ".csv",
                        encoding="UTF8",
                        index=False,
                    )
                except Exception:
                    pass


def _decoration_(func: Callable):
    """
    - func.__module__ 만으로 daily/weekly 구분해서 str_instance 결정 (요청 조건 준수)
    - daily_netting -> PYNettingODSD
    - weekly_netting -> PYNettingODSW
    """
    @wraps(func)
    def wrapper(*args, **kwargs):
        # 0) logger / output dir 준비
        str_instance = _get_str_instance_from_module(func)
        logger = _get_logger(str_instance)
        is_local = common.gfn_get_isLocal()
        # str_output_dir = _get_output_dir(str_instance,func)
        _version = kwargs.get("lv_ParamVersion")
        str_output_dir = _get_output_dir_version(str_instance,_version)

        # 1) 함수 시작 시각
        tm_start = time.time()

        # 2) 함수 실행
        result = func(*args, **kwargs)

        # 3) 종료/시간 로그
        tm_end = time.time()
        logger.Note(
            p_note=f"[{func.__name__}] Total time is {tm_end - tm_start:.5f} sec.",
            p_log_level=common.G_log_level.debug(),
        )

        # 4) Step log 및 DF log/csv
        _step_no = kwargs.get("p_step_no")
        _step_desc = kwargs.get("p_step_desc")
        _df_name = kwargs.get("p_df_name")
        _warn_desc = kwargs.get("p_warn_desc")
        _exception_flag = kwargs.get("p_exception_flag")

        if _step_no is not None and _step_desc is not None:
            logger.Step(p_step_no=_step_no, p_step_desc=_step_desc)

        if _warn_desc is not None:
            if isinstance(result, pd.DataFrame) and result.empty:
                if _exception_flag is not None:
                    if _exception_flag == 0:
                        logger.Note(p_note=_warn_desc, p_log_level=common.G_log_level.warning())
                    elif _exception_flag == 1:
                        raise Exception(_warn_desc)

        if _df_name is not None:
            fn_log_dataframe(
                result,
                _df_name,
                logger=logger,
                str_output_dir=str_output_dir,
                is_local=is_local,
            )

        return result

    return wrapper


def as_str_df(df: DF) -> DF:
    """
    category 컬럼만 string 으로 변환 (숫자/float/int 컬럼은 그대로 둠)
    - merge key 안정화 목적
    - 결측치는 "nan" 문자열로 만들지 않고 <NA> 유지
    """
    if df is None or df.empty:
        return df
    out = df.copy()
    cat_cols = out.select_dtypes(include=["category"]).columns
    if len(cat_cols) > 0:
        # category -> pandas StringDtype (결측치 보존)
        for c in cat_cols:
            out[c] = out[c].astype("string")
