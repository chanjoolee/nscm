def convert_dt_to_week(series: pd.Series) -> pd.Series:
    '''
    pandas datetime 타입의 series를, object 타입의 series로 변환한다.
    이때 변환된 week의 형식은 YYYYWW (ex: 202507)이다.
    '''
    isocalendar = series.dt.isocalendar()
    return isocalendar.year.astype(str) + isocalendar.week.astype(str).str.zfill(2)

def is_log_flag_on(accessor: object, plan_step: str):
    df_plan_option = accessor.df_plan_option
    df_filtered = df_plan_option.loc[
        df_plan_option[PO_.PLANSTEP] == plan_step
    ]
    if len(df_filtered) == 0:
        return False
    
    return df_filtered.iloc[0][PO_.LOGFLAG] == 'Y'

def validate(
        df_mine:pd.DataFrame, df_answer:pd.DataFrame, list_key_column:list,
        list_column_to_compare:list, max_count:int, is_bulk_data:bool = False
    ) -> Tuple[np.ndarray, np.ndarray]:
    '''
    생성한 dataframe을 정답 dataframe과 비교하여 값이 다른 데이터를 출력한다.

    df_mine: 나의 output dataframe
    df_answer: 정답 dataframe
    list_key_column: 데이터 정렬을 위한 컬럼 리스트
    list_column_to_compare: 값을 비교하고 싶은 컬럼 리스트
    max_count: max_count 만큼 틀린 행을 출력, -1일 경우 전부 출력
    is_bulk_data: True이면 전체 data를 반환, False이면 list_column_to_compare의 열을 반환
    '''

    is_same = True
    row_count = df_mine.shape[0]
    column_count = len(list_column_to_compare)
    
    if df_mine.shape[0] != df_answer.shape[0]:
        print(f'Row 개수 불일치(mine: {df_mine.shape[0]}, answer: {df_answer.shape[0]}): row수가 적은 data 기준으로 검증 실행\n')
        row_count = min(df_mine.shape[0], df_answer.shape[0])
        is_same = False
    
    df_mine_bulk = df_mine.sort_values(by=list_key_column)
    df_answer_bulk = df_answer.sort_values(by=list_key_column)


    df_mine = df_mine_bulk[list_column_to_compare].fillna(0)
    df_answer = df_answer_bulk[list_column_to_compare].fillna(0)

    nparr_mine = df_mine.to_numpy()
    nparr_answer = df_answer.to_numpy()

    count = 0
    for c in range(column_count):
        for r in range(row_count):
            if nparr_answer[r][c] != nparr_mine[r][c]:
                if count < max_count:
                    print(f'row: {r}, column: {list_column_to_compare[c]}(idx={c}), ', end='')
                    print(f'Mine: {nparr_mine[r][c]} {type(nparr_mine[r][c])}, ', end='')
                    print(f'Answer: {nparr_answer[r][c]} {type(nparr_answer[r][c])}')
                    count += 1
                is_same = False

    if is_same:
        print('검증 성공, 동일 데이터')
    else:
        print('검증 실패, 데이터 불일치')
    
    if is_bulk_data:
        return df_mine_bulk.to_numpy(), df_answer_bulk.to_numpy()
    return nparr_mine, nparr_answer

def overrange_to_datetime(series: pd.Series) -> pd.Series:
    # object가 아니면 pass
    if series.dtype != np.dtype('O'):
        return series
    
    nan_val_num = -11111111111
    nan_date = np.datetime64(nan_val_num, 'D')
    nat = np.datetime64('NaT')
    pd_date_min = np.datetime64('1677-09-22', 'D')
    pd_date_max = np.datetime64('2262-04-11', 'D')
    
    np_date = series.fillna(nan_val_num).to_numpy().astype('datetime64[D]')
    np_date = np.where(np_date == nan_date, nat, np_date)

    clipped_date = np.clip(np_date, pd_date_min, pd_date_max)

    return pd.to_datetime(clipped_date)

def convert_to_iyyyiw(date_str) -> str:
    '''
    Convert a date string in 'YYYY-MM-DD' format to IYYYIW format.
    
    Parameters:
    date_str (str): The date string to convert.
    
    Returns:
    str: The date in IYYYIW format.
    '''
    # 문자열을 날짜로 변환
    date_obj = datetime.strptime(date_str, '%Y-%m-%d')
    
    # ISO 연도와 주 번호 가져오기
    iso_year, iso_week, _ = date_obj.isocalendar()
    
    # IYYYIW 형식으로 변환
    return f'{iso_year}{iso_week:02d}'

def make_sequential_id(prefix: str, digit: int, dataframe: pd.DataFrame) -> pd.Series:
    return prefix + (dataframe.index + 1).astype(str).str.zfill(digit)
